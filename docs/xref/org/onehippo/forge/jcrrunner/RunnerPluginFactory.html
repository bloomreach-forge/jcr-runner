16  package org.onehippo.forge.jcrrunner;
17  
18  import java.io.FileNotFoundException;
19  import java.io.IOException;
20  
21  import org.slf4j.Logger;
22  import org.slf4j.LoggerFactory;
23  
24  import bsh.EvalError;
25  import bsh.TargetError;
26  
27  public final class RunnerPluginFactory {
28  
29      private static final Logger log = LoggerFactory.getLogger(RunnerPluginFactory.class);
30  
31      private static final String KEY_SCRIPT = "bsh";
32      private static final String KEY_CLASS = "class";
33  
34      private RunnerPluginFactory() {
35      }
36  
37      /**
38       * Instantiate a plugin from the beanshell
39       * @param config the configuration of the plugin
40       * @return a new Instance of the RunnerPlugin
41       */
42      public static RunnerPlugin createBeanShellPlugin(final RunnerPluginConfig config) {
43          try {
44              if (config.getValue(KEY_SCRIPT) == null || "".equals(config.getValue(KEY_SCRIPT))) {
45                  log.error(KEY_SCRIPT + " parameter not found. Not loading plugin: " + config.getId());
46                  return null;
47              }
48              RunnerPlugin plugin =(RunnerPlugin) new bsh.Interpreter().source(config.getValue(KEY_SCRIPT));
49              plugin.setId(config.getId());
50              plugin.setConfig(config);
51              return plugin;
52          } catch (TargetError e) {
53              log.error("The script or code of plugin " + config.getId() + "  threw an exception: " + e.getTarget());
54          } catch (EvalError e) {
55              log.error("There was an error in evaluating the script of plugin " + config.getId() + ":", e);
56          } catch (FileNotFoundException e) {
57              log.error("Script not found: " + config.getValue(KEY_SCRIPT) + ". Not loading plugin: " + config.getId());
58          } catch (IOException e) {
59              log.error("Error while trying to read script: " + config.getValue(KEY_SCRIPT) + "'" + e.getMessage() + "'"
60                      + " Not loading plugin: " + config.getId());
61          }
62          return null;
63      }
64  
65      /**
66       * Instantiate a plugin from the full className
67       * @param config the configuration of the plugin
68       * @return a new Instance of the RunnerPlugin
69       */
70      public static RunnerPlugin createJavaPlugin(final RunnerPluginConfig config) {
71          try {
72              if (config.getValue(KEY_CLASS) == null || "".equals(config.getValue(KEY_CLASS))) {
73                  log.error(KEY_CLASS + " parameter not found. Not loading plugin: " + config.getId());
74                  return null;
75              }
76              RunnerPlugin plugin = (RunnerPlugin) Class.forName(config.getValue(KEY_CLASS)).newInstance();
77              plugin.setId(config.getId());
78              plugin.setConfig(config);
79              return plugin;
80          } catch (ClassNotFoundException e) {
81              log.error("Class not found: " + config.getValue(KEY_CLASS) + ". Not loading plugin.");
82          } catch (InstantiationException e) {
83              log.error("Unable to instantiate class: " + config.getValue(KEY_CLASS) + ". Not loading plugin.");
84          } catch (IllegalAccessException e) {
85              log.error("Class not a RunnerPlugin: " + config.getValue(KEY_CLASS) + ". Not loading plugin.");
86          }
87          return null;
88      }
89  }
