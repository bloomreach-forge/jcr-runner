16  package org.onehippo.forge.jcrrunner;
17  
18  import java.io.IOException;
19  import java.io.InputStream;
20  import java.util.*;
21  import java.util.Map.Entry;
22  
23  public final class RunnerConfig {
24  
25      private static final String REPOSITORY_URL = "repository.url";
26      private static final String REPOSITORY_USER = "repository.user";
27      private static final String REPOSITORY_PASS = "repository.pass";
28  
29      private static final String PLUGINS_JAVA = "plugins.java";
30      private static final String PLUGINS_BEANSHELL = "plugins.beanshell";
31  
32      private SortedMap<String, RunnerPluginConfig> pluginConfigMap = new TreeMap<String, RunnerPluginConfig>();
33  
34      private String repositoryUrl;
35      private String repositoryUser;
36      private String repositoryPass;
37  
38      public String getRepositoryUrl() {
39          return repositoryUrl;
40      }
41  
42      public void setRepositoryUrl(String repositoryUrl) {
43          this.repositoryUrl = repositoryUrl;
44      }
45  
46      public String getRepositoryUser() {
47          return repositoryUser;
48      }
49  
50      public void setRepositoryUser(String repositoryUser) {
51          this.repositoryUser = repositoryUser;
52      }
53  
54      public String getRepositoryPass() {
55          return repositoryPass;
56      }
57  
58      public void setRepositoryPass(String repositoryPass) {
59          this.repositoryPass = repositoryPass;
60      }
61  
62      public List<RunnerPluginConfig> getPluginConfigs() {
63          return new ArrayList<RunnerPluginConfig>(pluginConfigMap.values());
64      }
65  
66      public RunnerConfig(InputStream inputStream) throws IOException {
67          Properties props = new Properties();
68          props.load(inputStream);
69          initializeRunnerConfig(props);
70      }
71  
72      public RunnerConfig(Properties props) {
73          initializeRunnerConfig(props);
74      }
75  
76      private void initializeRunnerConfig(Properties props) {
77          readRunnerConfig(props);
78          validateRunnerConfig();
79          readPluginInformation(props);
80      }
81  
82      private void readRunnerConfig(Properties props) {
83          setRepositoryUrl(props.getProperty(REPOSITORY_URL));
84          setRepositoryUser(props.getProperty(REPOSITORY_USER));
85          setRepositoryPass(props.getProperty(REPOSITORY_PASS));
86      }
87  
88      private void validateRunnerConfig() {
89          if (isEmpty(getRepositoryUrl())) {
90              throw new IllegalArgumentException(REPOSITORY_URL + " is missing.");
91          }
92          if (isEmpty(getRepositoryUser())) {
93              throw new IllegalArgumentException(REPOSITORY_USER + " is missing.");
94          }
95          if (isEmpty(getRepositoryPass())) {
96              throw new IllegalArgumentException(REPOSITORY_PASS + " is missing.");
97          }
98      }
99  
100     private boolean isEmpty(String s) {
101         return s == null || s.isEmpty();
102     }
103 
104     private void readPluginInformation(Properties props) {
105         Set<Entry<Object, Object>> entries = props.entrySet();
106         for (Entry<Object, Object> entry : entries) {
107             String key = (String) entry.getKey();
108             String value = (String) entry.getValue();
109             if (key.startsWith(PLUGINS_JAVA)) {
110                 int pos = key.indexOf('.', PLUGINS_JAVA.length() + 1);
111                 if (pos > 0) {
112                     String id = key.substring(PLUGINS_JAVA.length() + 1, pos);
113                     String pluginKey = key.substring(pos + 1);
114                     if (!pluginConfigMap.containsKey(id)) {
115                         pluginConfigMap.put(id, new RunnerPluginConfig(id, RunnerPluginType.JAVA));
116                     }
117                     pluginConfigMap.get(id).addEntry(pluginKey, value);
118                 }
119             }
120             if (key.startsWith(PLUGINS_BEANSHELL)) {
121                 int pos = key.indexOf('.', PLUGINS_BEANSHELL.length() + 1);
122                 if (pos > 0) {
123                     String id = key.substring(PLUGINS_BEANSHELL.length() + 1, pos);
124                     String pluginKey = key.substring(pos + 1);
125                     if (!pluginConfigMap.containsKey(id)) {
126                         pluginConfigMap.put(id, new RunnerPluginConfig(id, RunnerPluginType.BEANSHELL));
127                     }
128                     pluginConfigMap.get(id).addEntry(pluginKey, value);
129                 }
130             }
131         }
132     }
133 }
